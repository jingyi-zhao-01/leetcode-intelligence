# Problem 20. Valid Parentheses (correct)

def isValid(s):
    stack = []
    pairs = {')':'(', ']':'[', '}':'{'}
    for ch in s:
        if ch in '([{':
            stack.append(ch)
        else:
            if not stack or stack[-1] != pairs.get(ch, None):
                return False
            stack.pop()
    return not stack
